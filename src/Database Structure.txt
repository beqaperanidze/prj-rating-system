Table users {
  id integer [primary key]
  first_name varchar
  last_name varchar
  email varchar
  password varchar
  role varchar
  created_at timestamp
}

Table comments {
  id integer [primary key]
  message text
  author_id integer [ref: > users.id]
  seller_id integer [ref: > users.id]
  created_at timestamp
  approved boolean
}

Table game_objects {
  id integer [primary key]
  title varchar
  text text
  user_id integer [ref: > users.id]
  created_at timestamp
  updated_at timestamp
}
