
    users {
        int id PK
        varchar first_name
        varchar last_name
        varchar email
        varchar password
        varchar role
        timestamp created_at
        boolean approved
    }

        int id PK

        boolean approved
    }

    game_objects {
        int id PK
        varchar title
        text text
        int user_id FK
        timestamp created_at
        timestamp updated_at
    }

    ratings {
        int id PK
        int comment_id FK
        int rating_value
